import java.util.Scanner;

public class Reverse {


			public static void main(String[] args) {
			
				Scanner input = new Scanner(System.in);
				
				System.out.println("Enter a sentence: ");
				String sentence = input.nextLine();
				
				Scanner wordScanner = new Scanner(sentence);
				String finalResult = ""; 

		        while (wordScanner.hasNext()) {
		            String word = wordScanner.next();
		            String reversedWord = ""; 
		            
		            for (int i = word.length() - 1; i >= 0; i--) {
		                reversedWord = reversedWord + word.charAt(i);
		            }
		            
		            finalResult = finalResult + reversedWord + " ";
		        }

		        System.out.println(finalResult.trim());

		        wordScanner.close();
		        input.close();
							
			}

	

	}

		public class PunctuationMarks {

			public static void main(String[] args) {
				String text = "Mary had a little, her fleece was as white as snow, and everywhere Mary went, the lamb was sure to go. -that was a nice poem- the end.";
				
				int comma = 0;
				int period = 0;
				int hyphen = 0;
				
				for (int i = 0; i < text.length(); i++) {
					char ch = text.charAt(i);
					
					if (ch == ',') {
					    comma++;
					}
					else if (ch == '.') {
					    period++;
					}
					else if (ch == '-') {
					    hyphen++;
					}
					
				  }
				System.out.println("Punctuation Mark\tCount");
				System.out.println("-----------------------------");
				System.out.println(", (comma)\t\t" + comma);
				System.out.println(". (period)\t\t" + period);
				System.out.println("- (hyphen)\t\t" + hyphen);
			}
			

	}
